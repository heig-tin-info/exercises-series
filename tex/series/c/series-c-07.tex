\question Pour les exemples ci-dessous, indiquez successivement s'ils compilent sans erreur et le cas échéant, si les chaînes de caractères sont modifiables ou non.

\begin{parts}
\part
\begin{code}[c]
char s[] = 'foo';\end{code}
\answerline[Erreur, guillemet simple]

\part
\begin{code}[c]
char s[42] = "Les carottes sont cuites.";\end{code}
\answerline[L'expression compile et la chaîne est modifiable.]

\part
\begin{code}[c]
char s[2] = "Babel Fish";\end{code}
\answerline[Erreur : la taille de la chaîne est trop courte]

\part
\begin{code}[c]
char s[] = "foo\0bar\0";\end{code}
\answerline[L'expression compile et la chaîne est modificable]

\part
\begin{code}[c]
char *s = "Prix Béton 1981";\end{code}
\answerline[L'expression compile mais la chaîne est non modifiable]

\part
\begin{code}[c]
char *s[] = {"foo", "bar", "baz"};\end{code}
\answerline[Le tableau de chaînes est non modifiable]

\part
\begin{code}[c]
char s[] = {"foo" "bar" "baz"};\end{code}
\answerline[La chaîne compile et est modifiable. Elle vaut \canswer{"foobarbaz"}]

\end{parts}

\question Manipulation des chaînes de caractères : que retournent chacunes de ces fonctions ?
\begin{parts}

\part
\begin{code}[c]
int foo() { char s[] = "foo"; return strlen(s); }\end{code}
\answerline[3]

\part
\begin{code}[c]
int foo() { char s[] = "foo"; return sizeof(s); }\end{code}
\answerline[4]

\part
\begin{code}[c]
int foo() { char s[] = "foo"; return s[3]; }\end{code}
\answerline[0]

\part
\begin{code}[c]
int foo() { char s[] = "foo"; return s[4]; }\end{code}
\answerline[On ne peut pas savoir, peut être une \emph{segmentation fault}.]

\part
\begin{code}[c]
int foo() { char s[] = "foo\n\t\0bar"; return strlen(s); }
\end{code}
\answerline[5]

\end{parts}

\question En utilisant la bibliothèque \emph{ctype.h}, écrire une fonction qui transforme une\ chaîne de caractères reçue en majuscules. Le prototype de la fonction est \cinline!void majuscule(char *s)!.

\ifprintanswers

\begin{code}[c]
void majuscule(char *s) {
  while (*s) {
    *s = toupper(*s);
    s++;
  }
}
\end{code}
\fi

\question Soit un texte d'entrée, indiquer la position à laquelle se trouve une sous-chaîne, si elle existe. Sinon retourne -1. Le prototype de la fonction est \cinline!int position(char s[], char sub[])!. Aidez-vous également de la bibliothèque \emph{ctype.h}

\ifprintanswers

\begin{code}[c]
int position(char s[], char sub[]) {
  if (strstr(s, sub) == NULL) {
    return -1;
  }
  return strstr(s, sub) - s;
}
\end{code}
\fi

\question
\textbf{Difficile : } Il existe trois types de modifications qui peuvent être appliquées à une chaîne de caractère : insérer un caractère, supprimer un caractère, ou remplacer un caractère. Considérant deux chaînes de caractères, écrivez une fonction qui retourne si les deux chaînes ne sont qu'à une modification pr\`es.

\begin{itemize}
\item \cinline!pale, ple -> true!
\item \cinline!pales, pale -> true!
\item \cinline!pale, bale -> true!
\item \cinline!pale, bake -> false!
\end{itemize}

\ifprintanswers
\begin{solution}
\begin{code}[c]
bool one_away(char s1[], char s2[])
{
    int i = 0, j = 0, k = 0;
    while (s1[i] != '\0' && s2[j] != '\0')
    {
        if (s1[i] == s2[j])
        {
            i++;
            j++;
        }
        else if (s1[i] != s2[j])
        {
            if (s1[i + 1] == s2[j] || s1[i] == s2[j + 1])
            {
                i++;
                j++;
            }
            else
            {
                return 0;
            }
        }
    }
    if (s1[i] == '\0' && s2[j] == '\0')
    {
        return 1;
    }
    else if (s1[i] == '\0')
    {
        while (s2[j] != '\0')
        {
            if (s2[j + 1] == s2[j])
            {
                return 0;
            }
            j++;
        }
        return 1;
    }
    else if (s2[j] == '\0')
    {
        while (s1[i] != '\0')
        {
            if (s1[i + 1] == s1[i])
            {
                return 0;
            }
            i++;
        }
        return 1;
    }
    return 0;
}
\end{code}
\end{solution}

\fi
