int t[100];

for (int i = 0; i < 100; i++)
    t[i] = i + 1;

for (int m = 2; m < 10; m++)
    for (int i = 0; i < 100; i++)
        if (!(t[i] % m))
            t[i] = 0;

    if (t[i])
        printf("%d\n", t[i]);

\section{Passage par adresse}

Sans utiliser d'ordinateur, indiquer ce que le programme affiche à l'écran ?

\begin{lstlisting}
int foo(int* i, int *j, int *k, int p, int *q) {
    *i = *i + 2; // i
    j = i;
    *k += p;
    k = &p;
    *k += 4;
    q = k;
    return p;
}

int main(void) {
    int a = 11, b = 22, c =  33, d = 44, e = 55, f = d;
    e = foo(&a, &b, &c, d, &f);
    printf(
        "a. %d\n" "b. %d\n" "c. %d\n"
        "d. %d\n" "e. %d\n" "f. %d\n",
        a, b, c, d, e, f);
}
\end{lstlisting}
